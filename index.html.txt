// PASTE URL WEB APP BARU ANDA DI SINI
const API_URL = "https://script.google.com/macros/s/AKfycbwKC3n0qE0pzMHC77N6P2wG2SvSFTV2bbcchJxpxzLhS11FS5hCKosH_6vPrApzHm2-/exec"; 

// 1. KONTROL LOGIN
function executeLogin(event) {
  event.preventDefault();
  var pin = document.getElementById('pinInput').value;
  var btn = document.getElementById('btnLogin');
  var notif = document.getElementById('notifLogin');
  
  btn.disabled = true;
  btn.innerText = "MEMVERIFIKASI...";
  
  fetch(API_URL, {
    method: "POST",
    body: JSON.stringify({ aksi: "login", pin: pin })
  })
  .then(res => res.json())
  .then(respon => {
    if(respon.status) {
      notif.classList.add('hidden');
      document.getElementById('loginView').classList.add('hidden');
      document.getElementById('materiView').classList.remove('hidden');
      document.getElementById('namaStafDisplay').innerText = "STAF: " + respon.nama.toUpperCase();
      document.getElementById('sessionPin').value = pin;
      document.getElementById('sessionNama').value = respon.nama;
      pullDynamicMateri();
    } else {
      notif.innerText = respon.pesan;
      notif.className = "mt-4 font-bold text-center p-3 bg-red-100 text-red-800 border border-red-200 rounded-lg text-xs uppercase tracking-wide";
      notif.classList.remove('hidden');
      btn.disabled = false;
      btn.innerText = "VALIDASI AKSES";
    }
  });
}

// 2. AMBIL DATA MATERI
function pullDynamicMateri() {
  fetch(API_URL, {
    method: "POST",
    body: JSON.stringify({ aksi: "ambil_materi" })
  })
  .then(res => res.json())
  .then(respon => {
    buildMateriUI(respon.data);
  });
}

// 3. GENERATOR UI (Tetap sama seperti kode HTML dinamis sebelumnya)
function buildMateriUI(dataModul) {
  var judulDisplay = document.getElementById('judulModulDisplay');
  if (!dataModul || dataModul.length === 0) {
    judulDisplay.innerText = "BELUM ADA MATERI AKTIF";
    return;
  }
  var modul = dataModul[0]; 
  judulDisplay.innerText = "MODUL " + modul.id + ": " + modul.judul.toUpperCase();
  document.getElementById('videoPlayer').src = modul.videoUrl;
  document.getElementById('sessionModulId').value = modul.id;
  document.getElementById('teksSopDisplay').innerText = modul.materi ? modul.materi : "Ikuti instruksi video.";

  var htmlKuis = `
    <div class="bg-white p-4 border border-gray-200 rounded-lg shadow-sm">
      <label class="block font-bold text-sm text-gray-900 mb-3">${modul.soal1.tanya}</label>
      <div class="space-y-2">
        <label class="flex items-center text-sm p-2 rounded border border-gray-100 hover:bg-gray-50 cursor-pointer"><input type="radio" name="q1" value="${modul.soal1.opsiA}" required class="mr-3"> ${modul.soal1.opsiA}</label>
        <label class="flex items-center text-sm p-2 rounded border border-gray-100 hover:bg-gray-50 cursor-pointer"><input type="radio" name="q1" value="${modul.soal1.opsiB}"> ${modul.soal1.opsiB}</label>
        <label class="flex items-center text-sm p-2 rounded border border-gray-100 hover:bg-gray-50 cursor-pointer"><input type="radio" name="q1" value="${modul.soal1.opsiC}"> ${modul.soal1.opsiC}</label>
      </div>
    </div>
    <div class="bg-white p-4 border border-gray-200 rounded-lg shadow-sm mt-4">
      <label class="block font-bold text-sm text-gray-900 mb-3">${modul.soal2.tanya}</label>
      <div class="space-y-2">
        <label class="flex items-center text-sm p-2 rounded border border-gray-100 hover:bg-gray-50 cursor-pointer"><input type="radio" name="q2" value="${modul.soal2.opsiA}" required class="mr-3"> ${modul.soal2.opsiA}</label>
        <label class="flex items-center text-sm p-2 rounded border border-gray-100 hover:bg-gray-50 cursor-pointer"><input type="radio" name="q2" value="${modul.soal2.opsiB}"> ${modul.soal2.opsiB}</label>
        <label class="flex items-center text-sm p-2 rounded border border-gray-100 hover:bg-gray-50 cursor-pointer"><input type="radio" name="q2" value="${modul.soal2.opsiC}"> ${modul.soal2.opsiC}</label>
      </div>
    </div>
  `;
  document.getElementById('dynamicKuisArea').innerHTML = htmlKuis;
  document.getElementById('btnKuis').classList.remove('hidden');
}

// 4. SUBMIT JAWABAN KUIS
function executeKuisSubmit(event) {
  event.preventDefault();
  var btn = document.getElementById('btnKuis');
  var notif = document.getElementById('notifKuis');
  btn.disabled = true;
  btn.innerText = "MENGHITUNG SKOR KELULUSAN...";

  var payload = {
    aksi: "kuis",
    pin: document.getElementById('sessionPin').value,
    nama: document.getElementById('sessionNama').value,
    modulId: document.getElementById('sessionModulId').value,
    jawaban1: document.querySelector('input[name="q1"]:checked').value,
    jawaban2: document.querySelector('input[name="q2"]:checked').value
  };

  fetch(API_URL, {
    method: "POST",
    body: JSON.stringify(payload)
  })
  .then(res => res.json())
  .then(respon => {
    notif.classList.remove('hidden');
    if (respon.status === "LULUS") {
      notif.className = "mt-4 p-4 bg-green-50 text-green-800 border-green-200 rounded-lg text-xs font-semibold text-center";
      notif.innerText = respon.pesan;
      btn.classList.add('hidden');
    } else {
      notif.className = "mt-4 p-4 bg-red-50 text-red-800 border-red-200 rounded-lg text-xs font-semibold text-center";
      notif.innerText = respon.pesan;
      btn.disabled = false;
      btn.innerText = "KIRIM JAWABAN EVALUASI";
    }
  });
}
